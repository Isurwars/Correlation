592template <typename T> constexpr T determinant(const Matrix3<T> &m) noexcept {
593 return m[0][0] * (m[1][1] * m[2][2] - m[1][2] * m[2][1]) -
594 m[0][1] * (m[1][0] * m[2][2] - m[1][2] * m[2][0]) +
595 m[0][2] * (m[1][0] * m[2][1] - m[1][1] * m[2][0]);
596}

605template <typename T> constexpr Matrix3<T> invert(const Matrix3<T> &m) {
606 T det = determinant(m);
607 if (std::abs(det) < 1e-15)
608 throw std::runtime_error("singular matrix");
609 T inv = T{1} / det;
610
611 Vector3<T> c0{m[1][1] * m[2][2] - m[1][2] * m[2][1],
612 m[0][2] * m[2][1] - m[0][1] * m[2][2],
613 m[0][1] * m[1][2] - m[0][2] * m[1][1]};
614
615 Vector3<T> c1{m[1][2] * m[2][0] - m[1][0] * m[2][2],
616 m[0][0] * m[2][2] - m[0][2] * m[2][0],
617 m[0][2] * m[1][0] - m[0][0] * m[1][2]};
618
619 Vector3<T> c2{m[1][0] * m[2][1] - m[1][1] * m[2][0],
620 m[0][1] * m[2][0] - m[0][0] * m[2][1],
621 m[0][0] * m[1][1] - m[0][1] * m[1][0]};
622
623 return Matrix3<T>(c0 * inv, c1 * inv, c2 * inv);
624}
