105template <typename T> class Matrix3 {
106public:
107 using value_type = T;
108 // Constructors
109 constexpr Matrix3() noexcept { data_.fill(Vector3<T>{0, 0, 0}); }
110 constexpr Matrix3(const Vector3<T> &c0, const Vector3<T> &c1,
111 const Vector3<T> &c2) noexcept
112 : data_{c0, c1, c2} {}
113 // Column access
114 constexpr const Vector3<T> &operator[](std::size_t c) const noexcept {
115 return data_[c];
116 }
117 constexpr Vector3<T> &operator[](std::size_t c) noexcept { return data_[c]; }
118
119 // Element access (row, col)
120 constexpr T operator()(std::size_t r, std::size_t c) const noexcept {
121 return data_[c][r];
122 }
123 constexpr T &operator()(std::size_t r, std::size_t c) noexcept {
124 return data_[c][r];
125 }
126
127 constexpr std::array<std::array<T, 3>, 3> array() const noexcept {
128 return {{{data_[0][0], data_[1][0], data_[2][0]},
129 {data_[0][1], data_[1][1], data_[2][1]},
130 {data_[0][2], data_[1][2], data_[2][2]}}};
131 }
132
133 // Trace: sum of diagonal elements
134 constexpr T trace() const noexcept {
135 return data_[0][0] + data_[1][1] + data_[2][2];
136 }
137
138 // Scalar multiplication
139 constexpr Matrix3 operator*(T s) const noexcept {
140 return Matrix3(data_[0] * s, data_[1] * s, data_[2] * s);
141 }
142
143 // In-place scalar multiplication
144 constexpr Matrix3 &operator*=(T s) noexcept {
145 data_[0] = data_[0] * s;
146 data_[1] = data_[1] * s;
147 data_[2] = data_[2] * s;
148 return *this;
149 }
150
151 // In-place matrix addition
152 constexpr Matrix3 &operator+=(const Matrix3 &rhs) noexcept {
153 data_[0] += rhs.data_[0];
154 data_[1] += rhs.data_[1];
155 data_[2] += rhs.data_[2];
156 return *this;
157 }
158
159 // In-place matrix multiplication
160 constexpr Matrix3 &operator*=(const Matrix3 &rhs) noexcept {
161 const Matrix3 lhs = *this;
162 for (std::size_t j = 0; j < 3; ++j) {
163 data_[j] = lhs[0] * rhs(0, j) + lhs[1] * rhs(1, j) + lhs[2] * rhs(2, j);
164 }
165 return *this;
166 }
167
168private:
169 std::array<Vector3<T>, 3> data_;
170};

354template <typename T> constexpr T determinant(const Matrix3<T> &m) noexcept {
355 return m[0][0] * (m[1][1] * m[2][2] - m[1][2] * m[2][1]) -
356 m[0][1] * (m[1][0] * m[2][2] - m[1][2] * m[2][0]) +
357 m[0][2] * (m[1][0] * m[2][1] - m[1][1] * m[2][0]);
358}

367template <typename T> constexpr Matrix3<T> invert(const Matrix3<T> &m) {
368 T det = determinant(m);
369 if (std::abs(det) < 1e-15)
370 throw std::runtime_error("singular matrix");
371 T inv = T{1} / det;
372
373 Vector3<T> c0{m[1][1] * m[2][2] - m[1][2] * m[2][1],
374 m[0][2] * m[2][1] - m[0][1] * m[2][2],
375 m[0][1] * m[1][2] - m[0][2] * m[1][1]};
376
377 Vector3<T> c1{m[1][2] * m[2][0] - m[1][0] * m[2][2],
378 m[0][0] * m[2][2] - m[0][2] * m[2][0],
379 m[0][2] * m[1][0] - m[0][0] * m[1][2]};
380
381 Vector3<T> c2{m[1][0] * m[2][1] - m[1][1] * m[2][0],
382 m[0][1] * m[2][0] - m[0][0] * m[2][1],
383 m[0][0] * m[1][1] - m[0][1] * m[1][0]};
384
385 return Matrix3<T>(c0 * inv, c1 * inv, c2 * inv);
386}
