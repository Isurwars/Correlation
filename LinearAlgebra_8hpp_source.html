741template <typename T> [[nodiscard]] constexpr Matrix3<T> invert(const Matrix3<T> &matrix) {
742 const T det = determinant(matrix);
743 if (std::abs(det) < 1e-15) {
744 throw std::runtime_error("singular matrix");
745 }
746 const T inv = T{1} / det;
747
748 const Vector3<T> c_zero{matrix[1][1] * matrix[2][2] - matrix[1][2] * matrix[2][1],
749 matrix[0][2] * matrix[2][1] - matrix[0][1] * matrix[2][2],
750 matrix[0][1] * matrix[1][2] - matrix[0][2] * matrix[1][1]};
751
752 const Vector3<T> c_one{matrix[1][2] * matrix[2][0] - matrix[1][0] * matrix[2][2],
753 matrix[0][0] * matrix[2][2] - matrix[0][2] * matrix[2][0],
754 matrix[0][2] * matrix[1][0] - matrix[0][0] * matrix[1][2]};
755
756 const Vector3<T> c_two{matrix[1][0] * matrix[2][1] - matrix[1][1] * matrix[2][0],
757 matrix[0][1] * matrix[2][0] - matrix[0][0] * matrix[2][1],
758 matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0]};
759
760 return Matrix3<T>(c_zero * inv, c_one * inv, c_two * inv);
