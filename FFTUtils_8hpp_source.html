27inline void computeFFT(std::vector<std::complex<double>> &a, bool invert) {
28 size_t n = a.size();
29 if (n == 0)
30 return;
31
32 if ((n & (n - 1)) != 0) {
33 throw std::invalid_argument("FFT length must be a power of 2");
34 }
35
36 // Bit-reversal permutation (Optimized)
37 for (size_t i = 1, j = 0; i < n; i++) {
38 size_t bit = n >> 1;
39 for (; j & bit; bit >>= 1) {
40 j ^= bit;
41 }
42 j ^= bit;
43 if (i < j) {
44 std::swap(a[i], a[j]);
45 }
46 }
47
48 // Cooley-Tukey with precomputed twiddle basics
49 for (size_t len = 2; len <= n; len <<= 1) {
50 double angle = correlation::math::two_pi / len * (invert ? -1 : 1);
51 std::complex<double> wlen(std::cos(angle), std::sin(angle));
52 for (size_t i = 0; i < n; i += len) {
53 std::complex<double> w(1.0, 0.0);
54 for (size_t j = 0; j < len / 2; j++) {
55 std::complex<double> u = a[i + j];
56 std::complex<double> v = a[i + j + len / 2] * w;
57 a[i + j] = u + v;
58 a[i + j + len / 2] = u - v;
59 w *= wlen;
60 }
61 }
62 }
63
64 if (invert) {
65 for (auto &x : a) {
66 x /= static_cast<double>(n);
67 }
68 }
69}

81autocorrelate(const std::vector<double> &x,
82 std::vector<std::complex<double>> &workspace) {
83 const size_t n = x.size();
84 if (n == 0)
85 return {};
86
87 size_t len = 1;
88 while (len < 2 * n)
89 len <<= 1;
90
91 // Reuse caller-supplied workspace; only reallocates when length grows.
92 workspace.assign(len, {0.0, 0.0});
93 for (size_t i = 0; i < n; ++i)
94 workspace[i] = {x[i], 0.0};
95
96 computeFFT(workspace, false);
97
98 // In-place |X[k]|² — no intermediate vector needed.
99 for (size_t i = 0; i < len; ++i) {
100 const double re = workspace[i].real();
101 const double im = workspace[i].imag();
102 workspace[i] = {re * re + im * im, 0.0};
103 }
104
105 computeFFT(workspace, true);
106
107 std::vector<double> result(n);
108 for (size_t i = 0; i < n; ++i)
109 result[i] = workspace[i].real();
110 return result;
111}
