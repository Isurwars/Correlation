32inline bool isGoodFFTSize(size_t n) {
33 if (n == 0) {
34 return false;
35 }
36 while (n % 2 == 0) {
37 n /= 2;
38 }
39 while (n % 3 == 0) {
40 n /= 3;
41 }
42 while (n % 5 == 0) {
43 n /= 5;
44 }
45 return n == 1;
46}

51inline size_t findNextGoodFFTSize(size_t target) {
52#if defined(CORRELATION_USE_FFTW3) || defined(CORRELATION_USE_MKL)

148 size_t size = data.size();
149 if (size == 0) {
150 return;
151 }
152
153 thread_local MKLDescriptorCache cache;
154 auto &handles = invert ? cache.backward_handles : cache.forward_handles;
155 auto it = handles.find(size);
156 DFTI_DESCRIPTOR_HANDLE handle = nullptr;
157
158 if (it != handles.end()) {
159 handle = it->second;
160 } else {
161 MKL_LONG status = DftiCreateDescriptor(&handle, DFTI_DOUBLE, DFTI_COMPLEX, 1, static_cast<MKL_LONG>(size));
162 if (status != DFTI_NO_ERROR) {
163 throw std::runtime_error("MKL DftiCreateDescriptor failed");
164 }
165 if (invert) {
166 double scale = 1.0 / static_cast<double>(size);
167 DftiSetValue(handle, DFTI_BACKWARD_SCALE, scale);
168 }
169 status = DftiCommitDescriptor(handle);
170 if (status != DFTI_NO_ERROR) {
171 DftiFreeDescriptor(&handle);
172 throw std::runtime_error("MKL DftiCommitDescriptor failed");
173 }
174 handles[size] = handle;
175 }
176
177 MKL_LONG status;
178 if (invert) {
179 status = DftiComputeBackward(handle, data.data());
180 } else {
181 status = DftiComputeForward(handle, data.data());
182 }
183
184 if (status != DFTI_NO_ERROR) {
185 throw std::runtime_error("MKL DFTI computation failed");
186 }
187}

198inline void computeFFT(std::vector<std::complex<double>> &data, bool invert) {
199 size_t size = data.size();
200 if (size == 0) {
201 return;
202 }
203
204 if ((size & (size - 1)) != 0) {
205 throw std::invalid_argument("FFT length must be a power of 2");
206 }
207
208 // Bit-reversal permutation (Optimized)
209 for (size_t i = 1, j = 0; i < size; i++) {
210 size_t bit = size >> 1;
211 for (; (j & bit) != 0; bit >>= 1) {
212 j ^= bit;
213 }
214 j ^= bit;
215 if (i < j) {
216 std::swap(data[i], data[j]);
217 }
218 }
219
220 // Cooley-Tukey with precomputed twiddle basics
221 for (size_t len = 2; len <= size; len <<= 1) {
222 double angle = correlation::math::two_pi / static_cast<double>(len) * (invert ? -1 : 1);
223 std::complex<double> wlen(std::cos(angle), std::sin(angle));
224 for (size_t i = 0; i < size; i += len) {
225 std::complex<double> twiddle(1.0, 0.0);
226 for (size_t j = 0; j < len / 2; j++) {
227 std::complex<double> even_val = data[i + j];
228 std::complex<double> odd_val = data[i + j + len / 2] * twiddle;
229 data[i + j] = even_val + odd_val;
230 data[i + j + len / 2] = even_val - odd_val;
231 twiddle *= wlen;
232 }
233 }
234 }
235
236 if (invert) {
237 for (auto &val : data) {
238 val /= static_cast<double>(size);
239 }
240 }
241}

253inline std::vector<double> autocorrelate(const std::vector<double> &signal,
254 std::vector<std::complex<double>> &workspace) {
255 const size_t size = signal.size();
256 if (size == 0) {
257 return {};
258 }
259
260 size_t len = findNextGoodFFTSize(2 * size - 1);
261
262 // Reuse caller-supplied workspace; only reallocates when length grows.
263 workspace.assign(len, {0.0, 0.0});
264 for (size_t i = 0; i < size; ++i) {
265 workspace[i] = {signal[i], 0.0};
266 }
267
268 computeFFT(workspace, false);
269
270 // In-place |X[k]|² — no intermediate vector needed.
271 for (size_t i = 0; i < len; ++i) {
272 const double real_part = workspace[i].real();
273 const double imag_part = workspace[i].imag();
274 workspace[i] = {real_part * real_part + imag_part * imag_part, 0.0};
275 }
276
277 computeFFT(workspace, true);
278
279 std::vector<double> result(size);
280 for (size_t i = 0; i < size; ++i) {
281 result[i] = workspace[i].real();
282 }
283 return result;
284}
