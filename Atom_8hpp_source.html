153[[nodiscard]] inline double angle(const Atom &center, const Atom &a,
154 const Atom &b) noexcept {
155 const math::Vector3<double> vA = a.position() - center.position();
156 const math::Vector3<double> vB = b.position() - center.position();
157
158 const double norm_sq_A = math::dot(vA, vA);
159 const double norm_sq_B = math::dot(vB, vB);
160
161 if (norm_sq_A == 0.0 || norm_sq_B == 0.0) {
162 return 0.0;
163 }
164
165 double cos_theta = math::dot(vA, vB) / std::sqrt(norm_sq_A * norm_sq_B);
166
167 // Clamp for numerical stability
168 cos_theta = std::clamp(cos_theta, -1.0, 1.0);
169
170 return std::acos(cos_theta);
171}
