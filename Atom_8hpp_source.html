158[[nodiscard]] inline double angle(const Atom &center, const Atom &atom_a, const Atom &atom_b) noexcept {
159 const math::Vector3<double> vec_A = atom_a.position() - center.position();
160 const math::Vector3<double> vec_B = atom_b.position() - center.position();
161
162 const double norm_sq_A = math::dot(vec_A, vec_A);
163 const double norm_sq_B = math::dot(vec_B, vec_B);
164
165 if (norm_sq_A == 0.0 || norm_sq_B == 0.0) {
166 return 0.0;
167 }
168
169 double cos_theta = math::dot(vec_A, vec_B) / std::sqrt(norm_sq_A * norm_sq_B);
170
171 // Clamp for numerical stability
172 cos_theta = std::clamp(cos_theta, -1.0, 1.0);
173
174 return std::acos(cos_theta);
175}
