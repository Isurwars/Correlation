135[[nodiscard]] inline double angle(const Atom &center, const Atom &a,
136 const Atom &b) noexcept {
137 const correlation::math::Vector3<double> vA =
138 a.position() - center.position();
139 const correlation::math::Vector3<double> vB =
140 b.position() - center.position();
141
142 const double norm_sq_A = correlation::math::dot(vA, vA);
143 const double norm_sq_B = correlation::math::dot(vB, vB);
144
145 if (norm_sq_A == 0.0 || norm_sq_B == 0.0) {
146 return 0.0;
147 }
148
149 double cos_theta =
150 correlation::math::dot(vA, vB) / std::sqrt(norm_sq_A * norm_sq_B);
151
152 // Clamp for numerical stability
153 cos_theta = std::clamp(cos_theta, -1.0, 1.0);
154
155 return std::acos(cos_theta);
156}
