65inline double sph_legendre(int l, int m, double theta) {
66 if (m < 0 || m > l) {
67 return 0.0;
68 }
69
70 double x = std::cos(theta);
71
72 // Compute P_m^m(x)
73 double p_mm = 1.0;
74 if (m > 0) {
75 double somx2 = std::sqrt((1.0 - x) * (1.0 + x));
76 double fact = 1.0;
77 for (int i = 1; i <= m; ++i) {
78 p_mm *= -fact * somx2;
79 fact += 2.0;
80 }
81 }
82
83 double p_lm = p_mm;
84
85 if (l != m) {
86 // Compute P_{m+1}^m(x)
87 double p_mp1m = x * (2 * m + 1) * p_mm;
88 p_lm = p_mp1m;
89
90 if (l > m + 1) {
91 // Compute P_l^m(x) using recurrence relation
92 double p_ll_2 = p_mm; // P_{l-2}^m
93 double p_ll_1 = p_mp1m; // P_{l-1}^m
94
95 for (int ll = m + 2; ll <= l; ++ll) {
96 p_lm = (x * (2 * ll - 1) * p_ll_1 - (ll + m - 1) * p_ll_2) / (ll - m);
97 p_ll_2 = p_ll_1;
98 p_ll_1 = p_lm;
99 }
100 }
101 }
102
103 // Normalization factor
104 double norm = std::sqrt((2.0 * l + 1.0) / (4.0 * std::numbers::pi) *
105 factorial(l - m) / factorial(l + m));
106
107 // Cancel Condon-Shortley phase to match std::sph_legendre
108 if (m % 2 != 0) {
109 p_lm = -p_lm;
110 }
111
112 return p_lm * norm;
113}

125inline void sph_legendre_batch(int l, int m, const double *CORRELATION_RESTRICT theta,
126 double *CORRELATION_RESTRICT results, size_t count) {
127 if (m < 0 || m > l) {
128 for (size_t i = 0; i < count; ++i) {
129 results[i] = 0.0;
130 }
131 return;
132 }
133
134 // Precompute normalization factor and Condon-Shortley phase
135 double norm = std::sqrt((2.0 * l + 1.0) / (4.0 * std::numbers::pi) *
136 factorial(l - m) / factorial(l + m));
137 if (m % 2 != 0) {
138 norm = -norm;
139 }
140
141#if defined(CORRELATION_SIMD_AVX512)
142 size_t i = 0;
143 const __m512d vnorm = _mm512_set1_pd(norm);
144 for (; i + 8 <= count; i += 8) {
145 double x_arr[8], somx2_arr[8];
146 for (int k = 0; k < 8; ++k) {
147 x_arr[k] = std::cos(theta[i + k]);
148 somx2_arr[k] = std::sqrt((1.0 - x_arr[k]) * (1.0 + x_arr[k]));
149 }
150 __m512d vx = _mm512_loadu_pd(x_arr);
151 __m512d vsomx2 = _mm512_loadu_pd(somx2_arr);
152
153 __m512d vp_mm = _mm512_set1_pd(1.0);
154 if (m > 0) {
155 double fact = 1.0;
156 for (int k = 1; k <= m; ++k) {
157 vp_mm = _mm512_mul_pd(vp_mm, _mm512_mul_pd(_mm512_set1_pd(-fact), vsomx2));
158 fact += 2.0;
159 }
160 }
161
162 __m512d vp_lm = vp_mm;
163 if (l != m) {
164 __m512d vp_mp1m = _mm512_mul_pd(_mm512_mul_pd(vx, _mm512_set1_pd(2.0 * m + 1.0)), vp_mm);
165 vp_lm = vp_mp1m;
166
167 if (l > m + 1) {
168 __m512d vp_ll_2 = vp_mm;
169 __m512d vp_ll_1 = vp_mp1m;
170
171 for (int ll = m + 2; ll <= l; ++ll) {
172 __m512d term1 = _mm512_mul_pd(_mm512_mul_pd(vx, _mm512_set1_pd(2.0 * ll - 1.0)), vp_ll_1);
173 __m512d term2 = _mm512_mul_pd(_mm512_set1_pd(ll + m - 1.0), vp_ll_2);
174 __m512d diff = _mm512_sub_pd(term1, term2);
175 vp_lm = _mm512_mul_pd(diff, _mm512_set1_pd(1.0 / (ll - m)));
176 vp_ll_2 = vp_ll_1;
177 vp_ll_1 = vp_lm;
178 }
179 }
180 }
181
182 __m512d vres = _mm512_mul_pd(vp_lm, vnorm);
183 _mm512_storeu_pd(results + i, vres);
184 }
185 for (; i < count; ++i) {
186 results[i] = sph_legendre(l, m, theta[i]);
187 }
188#elif defined(CORRELATION_SIMD_AVX2)
189 size_t i = 0;
190 const __m256d vnorm = _mm256_set1_pd(norm);
191 for (; i + 4 <= count; i += 4) {
192 double x_arr[4], somx2_arr[4];
193 for (int k = 0; k < 4; ++k) {
194 x_arr[k] = std::cos(theta[i + k]);
195 somx2_arr[k] = std::sqrt((1.0 - x_arr[k]) * (1.0 + x_arr[k]));
196 }
197 __m256d vx = _mm256_loadu_pd(x_arr);
198 __m256d vsomx2 = _mm256_loadu_pd(somx2_arr);
199
200 __m256d vp_mm = _mm256_set1_pd(1.0);
201 if (m > 0) {
202 double fact = 1.0;
203 for (int k = 1; k <= m; ++k) {
204 vp_mm = _mm256_mul_pd(vp_mm, _mm256_mul_pd(_mm256_set1_pd(-fact), vsomx2));
205 fact += 2.0;
206 }
207 }
208
209 __m256d vp_lm = vp_mm;
210 if (l != m) {
211 __m256d vp_mp1m = _mm256_mul_pd(_mm256_mul_pd(vx, _mm256_set1_pd(2.0 * m + 1.0)), vp_mm);
212 vp_lm = vp_mp1m;
213
214 if (l > m + 1) {
215 __m256d vp_ll_2 = vp_mm;
216 __m256d vp_ll_1 = vp_mp1m;
217
218 for (int ll = m + 2; ll <= l; ++ll) {
219 __m256d term1 = _mm256_mul_pd(_mm256_mul_pd(vx, _mm256_set1_pd(2.0 * ll - 1.0)), vp_ll_1);
220 __m256d term2 = _mm256_mul_pd(_mm256_set1_pd(ll + m - 1.0), vp_ll_2);
221 __m256d diff = _mm256_sub_pd(term1, term2);
222 vp_lm = _mm256_mul_pd(diff, _mm256_set1_pd(1.0 / (ll - m)));
223 vp_ll_2 = vp_ll_1;
224 vp_ll_1 = vp_lm;
225 }
226 }
227 }
228
229 __m256d vres = _mm256_mul_pd(vp_lm, vnorm);
230 _mm256_storeu_pd(results + i, vres);
231 }
232 for (; i < count; ++i) {
233 results[i] = sph_legendre(l, m, theta[i]);
234 }

238 }
