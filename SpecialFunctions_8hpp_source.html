75inline double sph_legendre(LegendreParams params, double theta) {
76 const int degree = params.degree;
77 const int order = params.order;
78 if (order < 0 || order > degree) {
79 return 0.0;
80 }
81
82 double cos_theta = std::cos(theta);
83
84 // Compute P_m^m(x)
85 double p_mm = 1.0;
86 if (order > 0) {
87 double somx2 = std::sqrt((1.0 - cos_theta) * (1.0 + cos_theta));
88 double fact = 1.0;
89 for (int i = 1; i <= order; ++i) {
90 p_mm *= -fact * somx2;
91 fact += 2.0;
92 }
93 }
94
95 double p_lm = p_mm;
96
97 if (degree != order) {
98 // Compute P_{m+1}^m(x)
99 double p_mp1m = cos_theta * (2 * order + 1) * p_mm;
100 p_lm = p_mp1m;
101
102 if (degree > order + 1) {
103 // Compute P_l^m(x) using recurrence relation
104 double p_ll_2 = p_mm; // P_{l-2}^m
105 double p_ll_1 = p_mp1m; // P_{l-1}^m
106
107 for (int ll = order + 2; ll <= degree; ++ll) {
108 p_lm = (cos_theta * (2 * ll - 1) * p_ll_1 - (ll + order - 1) * p_ll_2) / (ll - order);
109 p_ll_2 = p_ll_1;
110 p_ll_1 = p_lm;
111 }
112 }
113 }
114
115 // Normalization factor
116 double norm = std::sqrt((2.0 * degree + 1.0) / (4.0 * correlation::math::pi) * factorial(degree - order) /
117 factorial(degree + order));
118
119 // Cancel Condon-Shortley phase to match std::sph_legendre
120 if (order % 2 != 0) {
121 p_lm = -p_lm;
122 }
123
124 return p_lm * norm;
125}

136inline void sph_legendre_batch(LegendreParams params, const double *CORRELATION_RESTRICT theta,
137 double *CORRELATION_RESTRICT results, size_t count) {
138 const int degree = params.degree;
139 const int order = params.order;
140 if (order < 0 || order > degree) {
141 for (size_t idx = 0; idx < count; ++idx) {
142 results[idx] = 0.0;
143 }
144 return;
145 }
146
147 // Precompute normalization factor and Condon-Shortley phase
148 double norm = std::sqrt((2.0 * degree + 1.0) / (4.0 * correlation::math::pi) * factorial(degree - order) /
149 factorial(degree + order));
150 if (order % 2 != 0) {
151 norm = -norm;
152 }
153
154#if defined(CORRELATION_SIMD_AVX512)
155 size_t idx = 0;
156 const __m512d vnorm = _mm512_set1_pd(norm);
157 for (; idx + 8 <= count; idx += 8) {
158 std::array<double, 8> x_arr{};
159 std::array<double, 8> somx2_arr{};
160 for (size_t k = 0; k < 8; ++k) {
161 x_arr.at(k) = std::cos(theta[idx + k]);
162 somx2_arr.at(k) = std::sqrt((1.0 - x_arr.at(k)) * (1.0 + x_arr.at(k)));
163 }
164 __m512d v_x = _mm512_loadu_pd(x_arr.data());
165 __m512d v_somx2 = _mm512_loadu_pd(somx2_arr.data());
166
167 __m512d vp_mm = _mm512_set1_pd(1.0);
168 if (order > 0) {
169 double fact = 1.0;
170 for (int k = 1; k <= order; ++k) {
171 vp_mm = _mm512_mul_pd(vp_mm, _mm512_mul_pd(_mm512_set1_pd(-fact), v_somx2));
172 fact += 2.0;
173 }
174 }
175
176 __m512d vp_lm = vp_mm;
177 if (degree != order) {
178 __m512d vp_mp1m = _mm512_mul_pd(_mm512_mul_pd(v_x, _mm512_set1_pd(2.0 * order + 1.0)), vp_mm);
179 vp_lm = vp_mp1m;
180
181 if (degree > order + 1) {
182 __m512d vp_ll_2 = vp_mm;
183 __m512d vp_ll_1 = vp_mp1m;
184
185 for (int ll = order + 2; ll <= degree; ++ll) {
186 __m512d term1 = _mm512_mul_pd(_mm512_mul_pd(v_x, _mm512_set1_pd(2.0 * ll - 1.0)), vp_ll_1);
187 __m512d term2 = _mm512_mul_pd(_mm512_set1_pd(ll + order - 1.0), vp_ll_2);
188 __m512d diff = _mm512_sub_pd(term1, term2);
189 vp_lm = _mm512_mul_pd(diff, _mm512_set1_pd(1.0 / (ll - order)));
190 vp_ll_2 = vp_ll_1;
191 vp_ll_1 = vp_lm;
192 }
193 }
194 }
195
196 __m512d vres = _mm512_mul_pd(vp_lm, vnorm);
197 _mm512_storeu_pd(results + idx, vres);
198 }
199 for (; idx < count; ++idx) {
200 results[idx] = sph_legendre({.degree = degree, .order = order}, theta[idx]);
201 }
202#elif defined(CORRELATION_SIMD_AVX2)
203 size_t idx = 0;
204 const __m256d vnorm = _mm256_set1_pd(norm);
205 for (; idx + 4 <= count; idx += 4) {
206 std::array<double, 4> x_arr{};
207 std::array<double, 4> somx2_arr{};
208 for (size_t k = 0; k < 4; ++k) {
209 x_arr.at(k) = std::cos(theta[idx + k]);

211 }
212 __m256d v_x = _mm256_loadu_pd(x_arr.data());
213 __m256d v_somx2 = _mm256_loadu_pd(somx2_arr.data());
214
215 __m256d vp_mm = _mm256_set1_pd(1.0);
216 if (order > 0) {
217 double fact = 1.0;
218 for (int k = 1; k <= order; ++k) {
219 vp_mm = _mm256_mul_pd(vp_mm, _mm256_mul_pd(_mm256_set1_pd(-fact), v_somx2));
220 fact += 2.0;
221 }
222 }
223
224 __m256d vp_lm = vp_mm;
225 if (degree != order) {
226 __m256d vp_mp1m = _mm256_mul_pd(_mm256_mul_pd(v_x, _mm256_set1_pd(2.0 * order + 1.0)), vp_mm);
227 vp_lm = vp_mp1m;
228
229 if (degree > order + 1) {
230 __m256d vp_ll_2 = vp_mm;
231 __m256d vp_ll_1 = vp_mp1m;

235 __m256d term2 = _mm256_mul_pd(_mm256_set1_pd(ll + order - 1.0), vp_ll_2);
