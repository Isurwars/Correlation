46inline double dist_sq_scalar(double ax, double ay, double az, double bx,
47 double by, double bz) noexcept {
48 double dx = bx - ax;
49 double dy = by - ay;
50 double dz = bz - az;
51 return dx * dx + dy * dy + dz * dz;
52}

69inline void compute_dsq_block(double ax, double ay, double az,
70 const PositionBlock &block,
71 double *CORRELATION_RESTRICT out_dsq) noexcept {
72 const __m512d va_x = _mm512_set1_pd(ax);
73 const __m512d va_y = _mm512_set1_pd(ay);
74 const __m512d va_z = _mm512_set1_pd(az);
75
76 std::size_t k = 0;
77 for (; k + 8 <= block.count; k += 8) {
78 __m512d dx = _mm512_sub_pd(_mm512_loadu_pd(block.x + k), va_x);
79 __m512d dy = _mm512_sub_pd(_mm512_loadu_pd(block.y + k), va_y);
80 __m512d dz = _mm512_sub_pd(_mm512_loadu_pd(block.z + k), va_z);
81 __m512d dsq =
82 _mm512_fmadd_pd(dx, dx, _mm512_fmadd_pd(dy, dy, _mm512_mul_pd(dz, dz)));
83 _mm512_storeu_pd(out_dsq + k, dsq);
84 }
85 if (k < block.count) {
86 __mmask8 mask = static_cast<__mmask8>((1u << (block.count - k)) - 1u);
87 __m512d dx = _mm512_sub_pd(_mm512_maskz_loadu_pd(mask, block.x + k), va_x);
88 __m512d dy = _mm512_sub_pd(_mm512_maskz_loadu_pd(mask, block.y + k), va_y);
89 __m512d dz = _mm512_sub_pd(_mm512_maskz_loadu_pd(mask, block.z + k), va_z);
90 __m512d dsq =
91 _mm512_fmadd_pd(dx, dx, _mm512_fmadd_pd(dy, dy, _mm512_mul_pd(dz, dz)));
92 _mm512_mask_storeu_pd(out_dsq + k, mask, dsq);
93 }
94}

96inline double sinc_integral(double Q,
97 const double *CORRELATION_RESTRICT integrand,
98 const double *CORRELATION_RESTRICT rbins,
99 double *CORRELATION_RESTRICT sinqr_scratch,
100 std::size_t count) noexcept {
101 for (std::size_t j = 0; j < count; ++j) {
102 sinqr_scratch[j] = std::sin(Q * rbins[j]);
103 }
104 __m512d vacc = _mm512_setzero_pd();
105 std::size_t j = 0;
106 for (; j + 8 <= count; j += 8) {
107 __m512d vi = _mm512_loadu_pd(integrand + j);
108 __m512d vs = _mm512_loadu_pd(sinqr_scratch + j);
109 vacc = _mm512_fmadd_pd(vi, vs, vacc);
110 }
111 double acc = _mm512_reduce_add_pd(vacc);
112 for (; j < count; ++j)
113 acc += integrand[j] * sinqr_scratch[j];
114 return acc;
115}

179 double acc = _mm_cvtsd_f64(sum1);
180 for (; j < count; ++j)
181 acc += integrand[j] * sinqr_scratch[j];
182 return acc;
183}

238inline double debye_sum(double Q, const double *CORRELATION_RESTRICT distances,
239 double *CORRELATION_RESTRICT scratch,
240 std::size_t count) noexcept {
241 if (Q < 1e-9)
242 return static_cast<double>(count);
243
244 // Pre-pass: compute sinc(Q*r) = sin(Q*r)/(Q*r) for every entry.
245 // The Taylor guard (x < 1e-4 → 1 - x²/6) prevents NaN for r = 0
246 // or near-zero distances before any vector arithmetic touches the data.
247 for (std::size_t j = 0; j < count; ++j) {
248 const double x = Q * distances[j];
249 scratch[j] = (x < 1e-4) ? (1.0 - x * x / 6.0) : (std::sin(x) / x);
250 }
251
252 // SIMD body simply reduces scratch[] — no division, no NaN risk.
253 __m512d vacc = _mm512_setzero_pd();
254 std::size_t j = 0;
255 for (; j + 8 <= count; j += 8)
256 vacc = _mm512_add_pd(vacc, _mm512_loadu_pd(scratch + j));
257 double acc = _mm512_reduce_add_pd(vacc);
258 for (; j < count; ++j)
259 acc += scratch[j];
260 return acc;
261}

335inline void normalize_rdf_bins(const double *CORRELATION_RESTRICT H,
336 const double *CORRELATION_RESTRICT rbins,
337 double g_norm, double inv_Ni_dr,
338 double inv_Nj_dr, double pi4_rho_j,
339 double *g_out, double *G_out, double *J_out,
340 double *Jinv_out, std::size_t count) noexcept {
341 const __m512d vg_norm = _mm512_set1_pd(g_norm);
342 const __m512d v1 = _mm512_set1_pd(1.0);
343 const __m512d vinNidr = _mm512_set1_pd(inv_Ni_dr);
344 const __m512d vinNjdr = _mm512_set1_pd(inv_Nj_dr);
345 const __m512d vpi4rho = _mm512_set1_pd(pi4_rho_j);
346
347 std::size_t k = 1;
348 for (; k + 8 <= count; k += 8) {
349 __m512d vH = _mm512_loadu_pd(H + k);
350 __m512d vr = _mm512_loadu_pd(rbins + k);
351 __m512d vr2 = _mm512_mul_pd(vr, vr);
352 __m512d vg = _mm512_div_pd(_mm512_mul_pd(vH, vg_norm), vr2);
353 _mm512_storeu_pd(g_out + k, vg);
354 _mm512_storeu_pd(
355 G_out + k,
356 _mm512_mul_pd(vpi4rho, _mm512_mul_pd(vr, _mm512_sub_pd(vg, v1))));
357 _mm512_storeu_pd(J_out + k, _mm512_mul_pd(vH, vinNidr));
358 _mm512_storeu_pd(Jinv_out + k, _mm512_mul_pd(vH, vinNjdr));
359 }
360 for (; k < count; ++k) {
361 const double r = rbins[k];
362 if (r < 1e-9)
363 continue;
364 const double g = H[k] * g_norm / (r * r);
365 g_out[k] = g;
366 G_out[k] = pi4_rho_j * r * (g - 1.0);
367 J_out[k] = H[k] * inv_Ni_dr;
368 Jinv_out[k] = H[k] * inv_Nj_dr;
369 }
370}

629inline void miller_phase_sum(const double *CORRELATION_RESTRICT c1,
630 const double *CORRELATION_RESTRICT s1,
631 const double *CORRELATION_RESTRICT c2,
632 const double *CORRELATION_RESTRICT s2,
633 const double *CORRELATION_RESTRICT c3,
634 const double *CORRELATION_RESTRICT s3,
635 std::size_t count, double &c_sum,
636 double &s_sum) noexcept {
637 __m512d vc_sum = _mm512_setzero_pd();
638 __m512d vs_sum = _mm512_setzero_pd();
639 std::size_t i = 0;
640 for (; i + 8 <= count; i += 8) {
641 __m512d vc1 = _mm512_loadu_pd(c1 + i);
642 __m512d vs1 = _mm512_loadu_pd(s1 + i);
643 __m512d vc2 = _mm512_loadu_pd(c2 + i);
644 __m512d vs2 = _mm512_loadu_pd(s2 + i);
645 __m512d vc3 = _mm512_loadu_pd(c3 + i);
646 __m512d vs3 = _mm512_loadu_pd(s3 + i);
647 __m512d vc12 = _mm512_fmsub_pd(vc1, vc2, _mm512_mul_pd(vs1, vs2));
648 __m512d vs12 = _mm512_fmadd_pd(vs1, vc2, _mm512_mul_pd(vc1, vs2));
649 __m512d vc123 = _mm512_fmsub_pd(vc12, vc3, _mm512_mul_pd(vs12, vs3));
650 __m512d vs123 = _mm512_fmadd_pd(vs12, vc3, _mm512_mul_pd(vc12, vs3));
651 vc_sum = _mm512_add_pd(vc_sum, vc123);
652 vs_sum = _mm512_add_pd(vs_sum, vs123);
653 }
654 c_sum = _mm512_reduce_add_pd(vc_sum);
655 s_sum = _mm512_reduce_add_pd(vs_sum);
656 for (; i < count; ++i) {
657 double c12 = c1[i] * c2[i] - s1[i] * s2[i];
658 double s12 = s1[i] * c2[i] + c1[i] * s2[i];
659 c_sum += c12 * c3[i] - s12 * s3[i];
660 s_sum += s12 * c3[i] + c12 * s3[i];
661 }

679inline void miller_phase_sum(const double *CORRELATION_RESTRICT c1,
680 const double *CORRELATION_RESTRICT s1,
681 const double *CORRELATION_RESTRICT c2,
682 const double *CORRELATION_RESTRICT s2,
683 const double *CORRELATION_RESTRICT c3,
684 const double *CORRELATION_RESTRICT s3,
685 std::size_t count, double &c_sum,
686 double &s_sum) noexcept {
687 __m256d vc_sum = _mm256_setzero_pd();
688 __m256d vs_sum = _mm256_setzero_pd();
689 std::size_t i = 0;
690 for (; i + 4 <= count; i += 4) {
691 __m256d vc1 = _mm256_loadu_pd(c1 + i);
692 __m256d vs1 = _mm256_loadu_pd(s1 + i);
693 __m256d vc2 = _mm256_loadu_pd(c2 + i);
694 __m256d vs2 = _mm256_loadu_pd(s2 + i);
695 __m256d vc3 = _mm256_loadu_pd(c3 + i);
696 __m256d vs3 = _mm256_loadu_pd(s3 + i);
697#if defined(__FMA__)
698 __m256d vc12 = _mm256_fmsub_pd(vc1, vc2, _mm256_mul_pd(vs1, vs2));
699 __m256d vs12 = _mm256_fmadd_pd(vs1, vc2, _mm256_mul_pd(vc1, vs2));
700 __m256d vc123 = _mm256_fmsub_pd(vc12, vc3, _mm256_mul_pd(vs12, vs3));
701 __m256d vs123 = _mm256_fmadd_pd(vs12, vc3, _mm256_mul_pd(vc12, vs3));
702#else
703 __m256d vc12 =
704 _mm256_sub_pd(_mm256_mul_pd(vc1, vc2), _mm256_mul_pd(vs1, vs2));
705 __m256d vs12 =
706 _mm256_add_pd(_mm256_mul_pd(vs1, vc2), _mm256_mul_pd(vc1, vs2));
707 __m256d vc123 =
708 _mm256_sub_pd(_mm256_mul_pd(vc12, vc3), _mm256_mul_pd(vs12, vs3));
709 __m256d vs123 =
710 _mm256_add_pd(_mm256_mul_pd(vs12, vc3), _mm256_mul_pd(vc12, vs3));
711#endif
712 vc_sum = _mm256_add_pd(vc_sum, vc123);
713 vs_sum = _mm256_add_pd(vs_sum, vs123);
714 }
715 __m128d clo = _mm256_castpd256_pd128(vc_sum);
716 __m128d chi = _mm256_extractf128_pd(vc_sum, 1);
717 __m128d c_sum2 = _mm_add_pd(clo, chi);
718 __m128d c_sum1 = _mm_hadd_pd(c_sum2, c_sum2);
719 c_sum = _mm_cvtsd_f64(c_sum1);
720 __m128d slo = _mm256_castpd256_pd128(vs_sum);
721 __m128d shi = _mm256_extractf128_pd(vs_sum, 1);
722 __m128d s_sum2 = _mm_add_pd(slo, shi);
723 __m128d s_sum1 = _mm_hadd_pd(s_sum2, s_sum2);
724 s_sum = _mm_cvtsd_f64(s_sum1);
725 for (; i < count; ++i) {
726 double c12 = c1[i] * c2[i] - s1[i] * s2[i];
727 double s12 = s1[i] * c2[i] + c1[i] * s2[i];
728 c_sum += c12 * c3[i] - s12 * s3[i];
729 s_sum += s12 * c3[i] + c12 * s3[i];
730 }
731}
