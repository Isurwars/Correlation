40inline double dist_sq_scalar(double ax, double ay, double az, double bx,
41 double by, double bz) noexcept {
42 double dx = bx - ax;
43 double dy = by - ay;
44 double dz = bz - az;
45 return dx * dx + dy * dy + dz * dz;
46}

63inline void compute_dsq_block(double ax, double ay, double az,
64 const PositionBlock &block,
65 double *CORRELATION_RESTRICT out_dsq) noexcept {
66 const __m512d va_x = _mm512_set1_pd(ax);
67 const __m512d va_y = _mm512_set1_pd(ay);
68 const __m512d va_z = _mm512_set1_pd(az);
69
70 std::size_t k = 0;
71 for (; k + 8 <= block.count; k += 8) {
72 __m512d dx = _mm512_sub_pd(_mm512_loadu_pd(block.x + k), va_x);
73 __m512d dy = _mm512_sub_pd(_mm512_loadu_pd(block.y + k), va_y);
74 __m512d dz = _mm512_sub_pd(_mm512_loadu_pd(block.z + k), va_z);
75 __m512d dsq =
76 _mm512_fmadd_pd(dx, dx, _mm512_fmadd_pd(dy, dy, _mm512_mul_pd(dz, dz)));
77 _mm512_storeu_pd(out_dsq + k, dsq);
78 }
79 if (k < block.count) {
80 __mmask8 mask = static_cast<__mmask8>((1u << (block.count - k)) - 1u);
81 __m512d dx = _mm512_sub_pd(_mm512_maskz_loadu_pd(mask, block.x + k), va_x);
82 __m512d dy = _mm512_sub_pd(_mm512_maskz_loadu_pd(mask, block.y + k), va_y);
83 __m512d dz = _mm512_sub_pd(_mm512_maskz_loadu_pd(mask, block.z + k), va_z);
84 __m512d dsq =
85 _mm512_fmadd_pd(dx, dx, _mm512_fmadd_pd(dy, dy, _mm512_mul_pd(dz, dz)));
86 _mm512_mask_storeu_pd(out_dsq + k, mask, dsq);
87 }
88}

90inline double sinc_integral(double Q,
91 const double *CORRELATION_RESTRICT integrand,
92 const double *CORRELATION_RESTRICT rbins,
93 double *CORRELATION_RESTRICT sinqr_scratch,
94 std::size_t count) noexcept {
95 for (std::size_t j = 0; j < count; ++j) {
96 sinqr_scratch[j] = std::sin(Q * rbins[j]);
97 }
98 __m512d vacc = _mm512_setzero_pd();
99 std::size_t j = 0;
100 for (; j + 8 <= count; j += 8) {
101 __m512d vi = _mm512_loadu_pd(integrand + j);
102 __m512d vs = _mm512_loadu_pd(sinqr_scratch + j);
103 vacc = _mm512_fmadd_pd(vi, vs, vacc);
104 }
105 double acc = _mm512_reduce_add_pd(vacc);
106 for (; j < count; ++j)
107 acc += integrand[j] * sinqr_scratch[j];
108 return acc;
109}

212inline double debye_sum(double Q, const double *CORRELATION_RESTRICT distances,
213 double *CORRELATION_RESTRICT scratch,
214 std::size_t count) noexcept {
215 if (Q < 1e-9)
216 return static_cast<double>(count);
217
218 // Pre-pass: compute sinc(Q*r) = sin(Q*r)/(Q*r) for every entry.
219 // The Taylor guard (x < 1e-4 → 1 - x²/6) prevents NaN for r = 0
220 // or near-zero distances before any vector arithmetic touches the data.
221 for (std::size_t j = 0; j < count; ++j) {
222 const double x = Q * distances[j];
223 scratch[j] = (x < 1e-4) ? (1.0 - x * x / 6.0) : (std::sin(x) / x);
224 }
225
226 // SIMD body simply reduces scratch[] — no division, no NaN risk.
227 __m512d vacc = _mm512_setzero_pd();
228 std::size_t j = 0;
229 for (; j + 8 <= count; j += 8)
230 vacc = _mm512_add_pd(vacc, _mm512_loadu_pd(scratch + j));
231 double acc = _mm512_reduce_add_pd(vacc);
232 for (; j < count; ++j)
233 acc += scratch[j];
234 return acc;
235}

309inline void normalize_rdf_bins(const double *CORRELATION_RESTRICT H,
310 const double *CORRELATION_RESTRICT rbins,
311 double g_norm, double inv_Ni_dr,
312 double inv_Nj_dr, double pi4_rho_j,
313 double *g_out, double *G_out, double *J_out,
314 double *Jinv_out, std::size_t count) noexcept {
315 const __m512d vg_norm = _mm512_set1_pd(g_norm);
316 const __m512d v1 = _mm512_set1_pd(1.0);
317 const __m512d vinNidr = _mm512_set1_pd(inv_Ni_dr);
318 const __m512d vinNjdr = _mm512_set1_pd(inv_Nj_dr);
319 const __m512d vpi4rho = _mm512_set1_pd(pi4_rho_j);
320
321 std::size_t k = 1;
322 for (; k + 8 <= count; k += 8) {
323 __m512d vH = _mm512_loadu_pd(H + k);
324 __m512d vr = _mm512_loadu_pd(rbins + k);
325 __m512d vr2 = _mm512_mul_pd(vr, vr);
326 __m512d vg = _mm512_div_pd(_mm512_mul_pd(vH, vg_norm), vr2);
327 _mm512_storeu_pd(g_out + k, vg);
328 _mm512_storeu_pd(
329 G_out + k,
330 _mm512_mul_pd(vpi4rho, _mm512_mul_pd(vr, _mm512_sub_pd(vg, v1))));
331 _mm512_storeu_pd(J_out + k, _mm512_mul_pd(vH, vinNidr));
332 _mm512_storeu_pd(Jinv_out + k, _mm512_mul_pd(vH, vinNjdr));
333 }
334 for (; k < count; ++k) {
335 const double r = rbins[k];
336 if (r < 1e-9)
337 continue;
338 const double g = H[k] * g_norm / (r * r);
339 g_out[k] = g;
340 G_out[k] = pi4_rho_j * r * (g - 1.0);
341 J_out[k] = H[k] * inv_Ni_dr;
342 Jinv_out[k] = H[k] * inv_Nj_dr;
343 }
344}

572#if defined(CORRELATION_SIMD_AVX512)

574inline void miller_phase_sum(const double *CORRELATION_RESTRICT c1,
575 const double *CORRELATION_RESTRICT s1,
576 const double *CORRELATION_RESTRICT c2,
577 const double *CORRELATION_RESTRICT s2,
578 const double *CORRELATION_RESTRICT c3,
579 const double *CORRELATION_RESTRICT s3,
580 std::size_t count, double &c_sum,
581 double &s_sum) noexcept {
582 __m512d vc_sum = _mm512_setzero_pd();
583 __m512d vs_sum = _mm512_setzero_pd();
584 std::size_t i = 0;
585 for (; i + 8 <= count; i += 8) {
586 __m512d vc1 = _mm512_loadu_pd(c1 + i);
587 __m512d vs1 = _mm512_loadu_pd(s1 + i);
588 __m512d vc2 = _mm512_loadu_pd(c2 + i);
589 __m512d vs2 = _mm512_loadu_pd(s2 + i);
590 __m512d vc3 = _mm512_loadu_pd(c3 + i);
591 __m512d vs3 = _mm512_loadu_pd(s3 + i);
592 __m512d vc12 = _mm512_fmsub_pd(vc1, vc2, _mm512_mul_pd(vs1, vs2));
593 __m512d vs12 = _mm512_fmadd_pd(vs1, vc2, _mm512_mul_pd(vc1, vs2));
594 __m512d vc123 = _mm512_fmsub_pd(vc12, vc3, _mm512_mul_pd(vs12, vs3));
595 __m512d vs123 = _mm512_fmadd_pd(vs12, vc3, _mm512_mul_pd(vc12, vs3));
596 vc_sum = _mm512_add_pd(vc_sum, vc123);
597 vs_sum = _mm512_add_pd(vs_sum, vs123);
598 }
599 c_sum = _mm512_reduce_add_pd(vc_sum);
600 s_sum = _mm512_reduce_add_pd(vs_sum);
601 for (; i < count; ++i) {
602 double c12 = c1[i] * c2[i] - s1[i] * s2[i];
603 double s12 = s1[i] * c2[i] + c1[i] * s2[i];
604 c_sum += c12 * c3[i] - s12 * s3[i];
605 s_sum += s12 * c3[i] + c12 * s3[i];

611inline void miller_phase_sum(const double *CORRELATION_RESTRICT c1,
612 const double *CORRELATION_RESTRICT s1,
613 const double *CORRELATION_RESTRICT c2,
614 const double *CORRELATION_RESTRICT s2,
615 const double *CORRELATION_RESTRICT c3,
616 const double *CORRELATION_RESTRICT s3,
617 std::size_t count, double &c_sum,
618 double &s_sum) noexcept {
619 __m256d vc_sum = _mm256_setzero_pd();
620 __m256d vs_sum = _mm256_setzero_pd();
621 std::size_t i = 0;
622 for (; i + 4 <= count; i += 4) {
623 __m256d vc1 = _mm256_loadu_pd(c1 + i);
624 __m256d vs1 = _mm256_loadu_pd(s1 + i);
625 __m256d vc2 = _mm256_loadu_pd(c2 + i);
626 __m256d vs2 = _mm256_loadu_pd(s2 + i);
627 __m256d vc3 = _mm256_loadu_pd(c3 + i);
628 __m256d vs3 = _mm256_loadu_pd(s3 + i);
629#if defined(__FMA__)
630 __m256d vc12 = _mm256_fmsub_pd(vc1, vc2, _mm256_mul_pd(vs1, vs2));
631 __m256d vs12 = _mm256_fmadd_pd(vs1, vc2, _mm256_mul_pd(vc1, vs2));
632 __m256d vc123 = _mm256_fmsub_pd(vc12, vc3, _mm256_mul_pd(vs12, vs3));
633 __m256d vs123 = _mm256_fmadd_pd(vs12, vc3, _mm256_mul_pd(vc12, vs3));
634#else
635 __m256d vc12 =
636 _mm256_sub_pd(_mm256_mul_pd(vc1, vc2), _mm256_mul_pd(vs1, vs2));
637 __m256d vs12 =
638 _mm256_add_pd(_mm256_mul_pd(vs1, vc2), _mm256_mul_pd(vc1, vs2));
639 __m256d vc123 =
640 _mm256_sub_pd(_mm256_mul_pd(vc12, vc3), _mm256_mul_pd(vs12, vs3));
641 __m256d vs123 =
642 _mm256_add_pd(_mm256_mul_pd(vs12, vc3), _mm256_mul_pd(vc12, vs3));
643#endif
644 vc_sum = _mm256_add_pd(vc_sum, vc123);
645 vs_sum = _mm256_add_pd(vs_sum, vs123);
646 }
647 __m128d clo = _mm256_castpd256_pd128(vc_sum);
648 __m128d chi = _mm256_extractf128_pd(vc_sum, 1);
649 __m128d c_sum2 = _mm_add_pd(clo, chi);
650 __m128d c_sum1 = _mm_hadd_pd(c_sum2, c_sum2);
651 c_sum = _mm_cvtsd_f64(c_sum1);
652 __m128d slo = _mm256_castpd256_pd128(vs_sum);
653 __m128d shi = _mm256_extractf128_pd(vs_sum, 1);
654 __m128d s_sum2 = _mm_add_pd(slo, shi);
655 __m128d s_sum1 = _mm_hadd_pd(s_sum2, s_sum2);
656 s_sum = _mm_cvtsd_f64(s_sum1);
657 for (; i < count; ++i) {
658 double c12 = c1[i] * c2[i] - s1[i] * s2[i];
659 double s12 = s1[i] * c2[i] + c1[i] * s2[i];
660 c_sum += c12 * c3[i] - s12 * s3[i];
661 s_sum += s12 * c3[i] + c12 * s3[i];
662 }
663}
