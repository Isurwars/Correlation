179inline double mapValue(double value, double data_min, double data_max, double svg_min, double svg_max) {
180 if (std::abs(data_max - data_min) < 1e-15) {
181 return (svg_min + svg_max) / 2.0;
182 }
183 return svg_min + (value - data_min) / (data_max - data_min) * (svg_max - svg_min);

194struct NiceScale {
195 double min = 0.0;
196 double max = 0.0;
197 double spacing = 0.0;
198 std::vector<double> ticks;
199
200 NiceScale() = default;
201
207 explicit NiceScale(const DataRange &range, int max_ticks = 6) {
208 double actual_min = range.min;
209 double actual_max = range.max;
210 if (std::abs(actual_max - actual_min) < 1e-12) {
211 min = actual_min - 0.5;
212 max = actual_min + 0.5;
213 spacing = 0.1;
214 } else {
215 double range_val = niceNum(actual_max - actual_min, false);
216 spacing = niceNum(range_val / (max_ticks - 1), true);
217 min = std::floor(actual_min / spacing) * spacing;
218 max = std::ceil(actual_max / spacing) * spacing;
219 }
220
221 double range_span = max - min;
222 int num_ticks = static_cast<int>(std::round(range_span / spacing)) + 1;
223 for (int idx = 0; idx < num_ticks; ++idx) {
224 double value = min + idx * spacing;
225 ticks.push_back(value);
226 }
227 }
228
229private:
236 static double niceNum(double range, bool round) {
237 double exponent = std::floor(std::log10(range));
238 double fraction = range / std::pow(10.0, exponent);
239 double nice_fraction = 0.0;
240
241 if (round) {
242 if (fraction < 1.5) {
243 nice_fraction = 1.0;
244 } else if (fraction < 3.0) {
245 nice_fraction = 2.0;
246 } else if (fraction < 7.0) {
247 nice_fraction = 5.0;
248 } else {
249 nice_fraction = 10.0;
250 }
251 } else {
252 if (fraction <= 1.0) {
253 nice_fraction = 1.0;
254 } else if (fraction <= 2.0) {
255 nice_fraction = 2.0;
256 } else if (fraction <= 5.0) {
257 nice_fraction = 5.0;
258 } else {
259 nice_fraction = 10.0;
260 }
261 }
262 return nice_fraction * std::pow(10.0, exponent);
263 }
264};

275inline std::string fmtScientific(double value) {
276 double abs_value = std::abs(value);
277 if (abs_value < 1e-12) {
278 return "0";
279 }
280
281 // Use scientific notation for very small/large values
282 if (abs_value < 0.001 || abs_value >= 10000.0) {
283 int exponent = static_cast<int>(std::floor(std::log10(abs_value)));
284 double fraction = value / std::pow(10.0, exponent);
285 std::string res = std::format("{:.1f}×10", fraction);
286 std::string exp_s = std::to_string(exponent);
287 for (char chr : exp_s) {
288 if (chr == '-') {
289 res += "⁻";
290 } else if (chr == '0') {
291 res += "⁰";
292 } else if (chr == '1') {
293 res += "¹";
294 } else if (chr == '2') {
295 res += "²";
296 } else if (chr == '3') {
297 res += "³";
298 } else if (chr == '4') {
299 res += "⁴";
300 } else if (chr == '5') {
301 res += "⁵";
302 } else if (chr == '6') {
303 res += "⁶";
304 } else if (chr == '7') {
305 res += "⁷";
306 } else if (chr == '8') {
307 res += "⁸";
308 } else if (chr == '9') {
309 res += "⁹";
310 }
311 }
312 return res;
313 }
314
315 // Otherwise use simple formatting
316 std::string str = std::format("{:.2f}", value);
317 auto dot = str.find('.');
318 if (dot != std::string::npos) {
319 std::size_t last = str.find_last_not_of('0');
320 if (last != std::string::npos && last > dot) {
321 str = str.substr(0, last + 1);
322 } else if (last == dot) {
323 str = str.substr(0, dot);
324 }
325 }
326 return str;
327}

390 void filterPartials() {
391 const auto &raw_partials = hist->smoothed_partials.empty() ? hist->partials : hist->smoothed_partials;
392 auto total_it = raw_partials.find("Total");
393 if (total_it != raw_partials.end()) {
394 partials["Total"] = total_it->second;
395 }
396
397 std::vector<std::pair<std::string, double>> candidates;
398 for (const auto &[key, value] : raw_partials) {
399 if (key == "Total" || key.starts_with("Frequency_")) {
400 continue;
401 }
402 double score = 0.0;
403 if (weights != nullptr && !weights->empty()) {
404 auto wit = weights->find(key);
405 if (wit != weights->end()) {
406 score = wit->second;
407 }
408 } else {
409 for (double val : value) {
410 score += std::abs(val);
411 }
412 }
413 candidates.emplace_back(key, score);
414 }
415
416 std::ranges::sort(candidates, [](const auto &lhs, const auto &rhs) { return lhs.second > rhs.second; });
417
418 std::size_t limit = std::min(candidates.size(), std::size_t(10));
419 for (std::size_t i = 0; i < limit; ++i) {
420 const std::string &key = candidates.at(i).first;
421 auto iter = raw_partials.find(key);
422 if (iter != raw_partials.end()) {
423 partials[key] = iter->second;
424 }
425 }
426 }

447 void computeScales() {
448 if (xs == nullptr || xs->empty()) {
449 return;
450 }
451 double raw_x_min = xs->front();
452 double raw_x_max = xs->back();
453 double raw_y_min = 0.0;
454 double raw_y_max = 0.0;
455 for (const auto &[key, value] : partials) {

458 raw_y_min = std::min(raw_y_min, val);

649 for (const auto &[key, value] : partials) {
650 std::size_t num_points = std::min(xs->size(), value.size());
651 for (std::size_t point_idx = 0; point_idx < num_points; ++point_idx) {
652 double sx_i = mapValue(xs->at(point_idx), xScale.min, xScale.max, px0, px1);
653 double sy_i = mapValue(value.at(point_idx), yScale.min, yScale.max, py1, py0);
654 double dx_i = sx_i - mouse_sx;
655 double dy_i = sy_i - mouse_sy;
656 double dist_sq = dx_i * dx_i + dy_i * dy_i;
657 if (dist_sq < min_dist_sq) {
658 min_dist_sq = dist_sq;
659 best.index = point_idx;
660 best.key = key;
661 }
662 }
663 }
664 return best;
665 }

673 void drawTooltipBox(const TooltipPosition &pos, const std::string &best_key,

692 "fill=\"{}\" fill-opacity=\"0.92\" stroke=\"{}\" stroke-width=\"1.5\" filter=\"url(#tooltip-shadow)\"/>\n",
693 tooltip_x, tooltip_y, tooltip_w, tooltip_h, card_bg, card_border);

695 // Header: x value (with unit or pure label)

712 col);

715 if (name == best_key) {

724 void drawHoverTooltip() {
725 if (hover == nullptr || !hover->active || hover->widget_width <= 0.0 || hover->widget_height <= 0.0 ||
726 xs == nullptr) {
727 return;
728 }
729 double svg_aspect = kW / kH;
730 double widget_aspect = hover->widget_width / hover->widget_height;
731 double scale = 1.0;
732 double offset_x = 0.0;
733 double offset_y = 0.0;
734
735 if (widget_aspect > svg_aspect) {
736 scale = hover->widget_height / kH;
737 offset_x = (hover->widget_width - kW * scale) / 2.0;
738 } else {
739 scale = hover->widget_width / kW;
740 offset_y = (hover->widget_height - kH * scale) / 2.0;
741 }
742
743 double mouse_sx = (hover->mouse_x - offset_x) / scale;
744 double mouse_sy = (hover->mouse_y - offset_y) / scale;
745
746 if (mouse_sx < px0 || mouse_sx > px1 || mouse_sy < py0 || mouse_sy > py1) {
747 return;
748 }
749
750 NearestPoint best = findNearestPoint(mouse_sx, mouse_sy);
751
752 std::size_t idx = best.index;
753 double target_x = xs->at(idx);
754 double sx_data = mapValue(target_x, xScale.min, xScale.max, px0, px1);
755
756 // Draw vertical guide line
757 svg << std::format(" <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
758 "stroke=\"{}\" stroke-width=\"1.5\" stroke-dasharray=\"4,4\"/>\n",
759 sx_data, py0, sx_data, py1, config->axis_color());
760
761 // Collect values and draw markers on curves
762 std::vector<std::tuple<std::string, double, std::string>> hover_values; // name, value, color
763 std::size_t color_idx = 0;
764 double snapped_sy_data = -1.0;
765 for (const auto &[key, value] : partials) {
766 if (idx < value.size()) {
767 const std::string col = color(color_idx++, config->palette);
768 double y_val = value.at(idx);
769 double sy_data = mapValue(y_val, yScale.min, yScale.max, py1, py0);
770
771 if (key == best.key) {
772 snapped_sy_data = sy_data;
773 }
774
775 // Bullet marker
776 svg << std::format(
777 " <circle cx=\"{:.1f}\" cy=\"{:.1f}\" r=\"6\" fill=\"{}\" stroke=\"{}\" stroke-width=\"2\"/>\n", sx_data,
778 sy_data, col, config->bg_color());
779
780 hover_values.emplace_back(key, y_val, col);
781 }
782 }
783
784 // Draw horizontal guide line to the snapped data point of the closest curve
785 if (snapped_sy_data >= py0 && snapped_sy_data <= py1) {
786 svg << std::format(" <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
787 "stroke=\"{}\" stroke-width=\"1.5\" stroke-dasharray=\"4,4\"/>\n",
788 px0, snapped_sy_data, sx_data, snapped_sy_data, config->axis_color());
789 }
790
791 drawTooltipBox({sx_data, snapped_sy_data, target_x}, best.key, hover_values);
792 }

794 std::string getResult() {
795 svg << "</svg>\n";
796 return svg.str();
797 }

1081 struct NearestPoint {
1082 std::size_t index = 0;
1083 std::string label;
1084 };
1085
1086 NearestPoint findNearestPoint(double sp_x, double sp_y) const {
1087 double min_dist_sq = 1e30;
1088 NearestPoint best;
1089
1090 for (const auto &dataset : *datasets) {
1091 const auto &partials =
1092 dataset.hist->smoothed_partials.empty() ? dataset.hist->partials : dataset.hist->smoothed_partials;
1093 auto pit = partials.find(*partial_key);
1094 if (pit == partials.end()) {
1095 continue;
1096 }
1097 const auto &x_values = dataset.hist->bins;
1098 const auto &y_values = pit->second;
1099 std::size_t n_points = std::min(x_values.size(), y_values.size());
1100 for (std::size_t i = 0; i < n_points; ++i) {
1101 double sp_x_i = detail::mapValue(x_values[i], xScale.min, xScale.max, px0, px1);
1102 double sp_y_i = detail::mapValue(y_values[i], yScale.min, yScale.max, py1, py0);
1103 double dx_i = sp_x_i - sp_x;
1104 double dy_i = sp_y_i - sp_y;
1105 double dist_sq = dx_i * dx_i + dy_i * dy_i;
1106 if (dist_sq < min_dist_sq) {
1107 min_dist_sq = dist_sq;
1108 best.index = i;
1109 best.label = dataset.label;
1110 }
1111 }
1112 }
1113 return best;
1114 }

1117 collectHoverValuesAndDrawMarkers(std::size_t idx, const std::string &best_label, double sx_data,
1118 double &snapped_sy_data) {
1119 std::vector<std::tuple<std::string, double, std::string>> hover_values;
1120 std::size_t color_idx = 0;
1121 for (const auto &dataset : *datasets) {
1122 const auto &partials =
1123 dataset.hist->smoothed_partials.empty() ? dataset.hist->partials : dataset.hist->smoothed_partials;
1124 auto pit = partials.find(*partial_key);
1125 if (pit != partials.end() && idx < pit->second.size()) {
1126 const std::string col = detail::color(color_idx++, config->palette);
1127 double y_val = pit->second[idx];
1128 double sy_data = detail::mapValue(y_val, yScale.min, yScale.max, py1, py0);

1201 void drawHoverTooltip() {
1202 if (!hover->active || hover->widget_width <= 0.0 || hover->widget_height <= 0.0) {
1203 return;
1204 }
1205 double svg_aspect = kWidth / kHeight;
1206 double widget_aspect = hover->widget_width / hover->widget_height;
1207 double scale = 1.0;
1208 double offset_x = 0.0;
1209 double offset_y = 0.0;
1210
1211 if (widget_aspect > svg_aspect) {
1212 scale = hover->widget_height / kHeight;
1213 offset_x = (hover->widget_width - kWidth * scale) / 2.0;
1214 } else {
1215 scale = hover->widget_width / kWidth;
1216 offset_y = (hover->widget_height - kHeight * scale) / 2.0;
1217 }
1218
1219 double sp_x = (hover->mouse_x - offset_x) / scale;
1220 double sp_y = (hover->mouse_y - offset_y) / scale;
1221
1222 if (sp_x < px0 || sp_x > px1 || sp_y < py0 || sp_y > py1) {
1223 return;
1224 }
1225
1226 auto [best_idx, best_label] = findNearestPoint(sp_x, sp_y);
1227
1228 const auto &x_bins = datasets->front().hist->bins;
1229 if (x_bins.empty() || best_idx >= x_bins.size()) {
1230 return;
1231 }
1232
1233 double target_x = x_bins[best_idx];
1234 double sx_data = detail::mapValue(target_x, xScale.min, xScale.max, px0, px1);
1235
1236 // Draw vertical guide line
1237 svg << std::format(" <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
1238 "stroke=\"{}\" stroke-width=\"1.5\" stroke-dasharray=\"4,4\"/>\n",
1239 sx_data, py0, sx_data, py1, config->axis_color());
1240
1241 double snapped_sy_data = -1.0;
1242 auto hover_values = collectHoverValuesAndDrawMarkers(best_idx, best_label, sx_data, snapped_sy_data);
1243
1244 // Draw horizontal guide line to the snapped data point of the closest dataset
1245 if (snapped_sy_data >= py0 && snapped_sy_data <= py1) {
1246 svg << std::format(" <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
1247 "stroke=\"{}\" stroke-width=\"1.5\" stroke-dasharray=\"4,4\"/>\n",
1248 px0, snapped_sy_data, sx_data, snapped_sy_data, config->axis_color());
1249 }
1250
1251 drawTooltipBox({sx_data, snapped_sy_data, target_x}, best_label, hover_values);
1252 }
