96inline double mapValue(double v, double data_min, double data_max,
97 double svg_min, double svg_max) {
98 if (std::abs(data_max - data_min) < 1e-15)
99 return (svg_min + svg_max) / 2.0;
100 return svg_min + (v - data_min) / (data_max - data_min) * (svg_max - svg_min);
101}

106struct NiceScale {
107 double min;
108 double max;
109 double spacing;
110 std::vector<double> ticks;
111
118 NiceScale(double actual_min, double actual_max, int max_ticks = 6) {
119 if (std::abs(actual_max - actual_min) < 1e-12) {
120 min = actual_min - 0.5;
121 max = actual_min + 0.5;
122 spacing = 0.1;
123 } else {
124 double range = niceNum(actual_max - actual_min, false);
125 spacing = niceNum(range / (max_ticks - 1), true);
126 min = std::floor(actual_min / spacing) * spacing;
127 max = std::ceil(actual_max / spacing) * spacing;
128 }
129
130 for (double v = min; v <= max + 0.5 * spacing; v += spacing) {
131 ticks.push_back(v);
132 }
133 }
134
135private:
142 double niceNum(double range, bool round) {
143 double exponent = std::floor(std::log10(range));
144 double fraction = range / std::pow(10.0, exponent);
145 double nice_fraction;
146
147 if (round) {
148 if (fraction < 1.5)
149 nice_fraction = 1.0;
150 else if (fraction < 3.0)
151 nice_fraction = 2.0;
152 else if (fraction < 7.0)
153 nice_fraction = 5.0;
154 else
155 nice_fraction = 10.0;
156 } else {
157 if (fraction <= 1.0)
158 nice_fraction = 1.0;
159 else if (fraction <= 2.0)
160 nice_fraction = 2.0;
161 else if (fraction <= 5.0)
162 nice_fraction = 5.0;
163 else
164 nice_fraction = 10.0;
165 }
166 return nice_fraction * std::pow(10.0, exponent);
167 }
168};

179inline std::string fmtScientific(double v) {
180 double abs_v = std::abs(v);
181 if (abs_v < 1e-12)
182 return "0";
183
184 // Use scientific notation for very small/large values
185 if (abs_v < 0.001 || abs_v >= 10000.0) {
186 int exponent = static_cast<int>(std::floor(std::log10(abs_v)));
187 double fraction = v / std::pow(10.0, exponent);
188 std::string res = std::format("{:.1f}×10", fraction);
189 std::string exp_s = std::to_string(exponent);
190 for (char c : exp_s) {
191 if (c == '-')
192 res += "⁻";
193 else if (c == '0')
194 res += "⁰";
195 else if (c == '1')
196 res += "¹";
197 else if (c == '2')
198 res += "²";
199 else if (c == '3')
200 res += "³";
201 else if (c == '4')
202 res += "⁴";
203 else if (c == '5')
204 res += "⁵";
205 else if (c == '6')
206 res += "⁶";
207 else if (c == '7')
208 res += "⁷";
209 else if (c == '8')
210 res += "⁸";
211 else if (c == '9')
212 res += "⁹";
213 }
214 return res;
215 }
216
217 // Otherwise use simple formatting
218 std::string s = std::format("{:.2f}", v);
219 auto dot = s.find('.');
220 if (dot != std::string::npos) {
221 std::size_t last = s.find_last_not_of('0');
222 if (last != std::string::npos && last > dot)
223 s = s.substr(0, last + 1);
224 else if (last == dot)
225 s = s.substr(0, dot);
226 }
227 return s;
228}

243inline std::string
244renderHistogramAsSvg(const correlation::analysis::Histogram &hist,
245 const PlotConfig &config = {}) {
246 std::string title = hist.title.empty() ? "Histogram" : hist.title;
247 std::string x_label = hist.x_label.empty() ? "x" : hist.x_label;
248 std::string y_label = hist.y_label.empty() ? "y" : hist.y_label;
249
250 // Add units if available
251 if (!hist.x_unit.empty())
252 x_label += std::format(" ({})", hist.x_unit);
253 if (!hist.y_unit.empty()) {
254 std::string y_unit = hist.y_unit;
255 // Standardize Å^-1 to Å⁻¹ for path rendering
256 if (y_unit == "Å^-1")
257 y_unit = "Å⁻¹";
258 y_label += std::format(" ({})", y_unit);
259 }
260
261 // ---- Layout configuration --------------------------------------------
262 const double kW = config.width;
263 const double kH = config.height;
264 // Margins (enough room for labels)
265 const double kLeft = 100.0;
266 const double kRight = 40.0;
267 const double kTop = 50.0;
268 const double kBot = 90.0;
269
270 const double px0 = kLeft;
271 const double px1 = kW - kRight;
272 const double py0 = kTop;
273 const double py1 = kH - kBot;
274
275 // ---- Data source -----------------------------------------------------
276 const auto &partials =
277 hist.smoothed_partials.empty() ? hist.partials : hist.smoothed_partials;
278 const auto &xs = hist.bins;
279
280 if (xs.empty() || partials.empty()) {
281 return std::format(
282 "<svg width='1200' height='900' xmlns=\"http://www.w3.org/2000/svg\" viewBox=\"0 0 {0} {1}\">"
283 "<rect width=\"100%\" height=\"100%\" fill=\"{2}\"/>"
284 "<path d=\"{3}\" fill=\"{4}\" stroke=\"none\"/></svg>",
285 kW, kH, config.bg_color(),
286 Roboto::instance().render("No data available", kW / 2.0, kH / 2.0 + 8.0,
287 24, "middle"),
288 config.text_color());
289 }
290
291 // ---- Compute ranges and nice ticks -----------------------------------
292 double raw_x_min = xs.front(), raw_x_max = xs.back();
293 double raw_y_min = 0.0, raw_y_max = 0.0;
294 for (const auto &[key, ys] : partials) {
295 for (double v : ys) {
296 raw_y_max = std::max(raw_y_max, v);
297 raw_y_min = std::min(raw_y_min, v);
298 }
299 }
300 // Add 5% padding to Y
301 double y_padding = (raw_y_max - raw_y_min) * 0.05;
302 raw_y_max += y_padding;
303
304 detail::NiceScale xScale(raw_x_min, raw_x_max, 11);
305 detail::NiceScale yScale(raw_y_min, raw_y_max, 8);
306
307 // ---- Build SVG -------------------------------------------------------
308 std::ostringstream svg;
309 svg << std::format(
310 "<svg width='1200' height='900' xmlns=\"http://www.w3.org/2000/svg\" viewBox=\"0 0 {} {}\">\n", kW,
311 kH);
312 svg << std::format(
313 " <rect width=\"100%\" height=\"100%\" fill=\"{}\" rx=\"6\"/>\n",
314 config.bg_color());
315
316 // Grid and Axes
317 svg << " <!-- Grid & Ticks -->\n";
318 for (double yv : yScale.ticks) {
319 double spy = detail::mapValue(yv, yScale.min, yScale.max, py1, py0);
320 // Grid line
321 if (config.show_grid) {
322 svg << std::format(
323 " <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
324 "stroke=\"{}\" stroke-width=\"0.8\" stroke-dasharray=\"3,3\"/>\n",
325 px0, spy, px1, spy, config.grid_color());
326 }
327 // Tick mark
328 svg << std::format(
329 " <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
330 "stroke=\"{}\" stroke-width=\"1.5\"/>\n",
331 px0 - 8.0, spy, px0, spy, config.axis_color());
332 // Label
333 svg << std::format(" <path d=\"{}\" fill=\"{}\" stroke=\"none\"/>\n",
334 Roboto::instance().render(detail::fmtScientific(yv),
335 px0 - 15.0, spy + 7.0, 20,
336 "end"),
337 config.text_color());
338 }
339
340 for (double xv : xScale.ticks) {
341 double spx = detail::mapValue(xv, xScale.min, xScale.max, px0, px1);
342 if (config.show_grid) {
343 svg << std::format(
344 " <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
345 "stroke=\"{}\" stroke-width=\"0.8\" stroke-dasharray=\"3,3\"/>\n",
346 spx, py0, spx, py1, config.grid_color());
347 }
348 svg << std::format(
349 " <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
350 "stroke=\"{}\" stroke-width=\"1.5\"/>\n",
351 spx, py1, spx, py1 + 8.0, config.axis_color());
352 // Label
353 svg << std::format(" <path d=\"{}\" fill=\"{}\" stroke=\"none\"/>\n",
354 Roboto::instance().render(detail::fmtScientific(xv), spx,
355 py1 + 25.0, 20, "middle"),
356 config.text_color());
357 }
358
359 // Draw axis border
360 svg << std::format(
361 " <rect x=\"{:.1f}\" y=\"{:.1f}\" width=\"{:.1f}\" height=\"{:.1f}\" "
362 "fill=\"none\" stroke=\"{}\" stroke-width=\"1.5\"/>\n",
363 px0, py0, px1 - px0, py1 - py0, config.axis_color());
364
365 // Emphasis lines for y=0 or y=1
366 std::vector<double> emphasis_values;
367 if (title.find("g(r)") != std::string::npos) {
368 emphasis_values.push_back(1.0);
369 } else if (title.find("G(r)") != std::string::npos) {
370 emphasis_values.push_back(0.0);
371 } else if (title.find("S(Q)") != std::string::npos ||
372 title.find("S(q)") != std::string::npos) {
373 emphasis_values.push_back(0.0);
374 emphasis_values.push_back(1.0);
375 }
376
377 for (double focus_y : emphasis_values) {
378 if (focus_y >= yScale.min && focus_y <= yScale.max) {
379 double spy = detail::mapValue(focus_y, yScale.min, yScale.max, py1, py0);
380 std::string extra =
381 (std::abs(focus_y - 1.0) < 1e-6) ? " stroke-dasharray=\"5,5\"" : "";
382 svg << std::format(
383 " <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
384 "stroke=\"{}\" stroke-width=\"1.8\"{}/>\n",
385 px0, spy, px1, spy, config.axis_color(), extra);
386 }
387 }
388
389 // Polylines
390 std::vector<std::pair<std::string, std::string>> legend;
391 std::size_t ci = 0;
392 for (const auto &[key, ys] : partials) {
393 const std::string col = detail::color(ci++);
394 svg << " <polyline fill=\"none\" stroke=\"" << col
395 << "\" stroke-width=\"3.0\" stroke-linejoin=\"round\" points=\"";
396 std::size_t n = std::min(xs.size(), ys.size());
397 for (std::size_t i = 0; i < n; ++i) {
398 double sx = detail::mapValue(xs[i], xScale.min, xScale.max, px0, px1);
399 double sy = detail::mapValue(ys[i], yScale.min, yScale.max, py1, py0);
400 svg << std::format("{:.2f},{:.2f} ", sx, sy);
401 }
402 svg << "\" />\n";
403 legend.push_back({key, col});
404 }
405
406 // Legend (Top Right)
407 double lx = px1 - 15.0;
408 double ly = py0 + 25.0;
409 for (auto it = legend.rbegin(); it != legend.rend(); ++it) {
410 svg << std::format(
411 " <line x1=\"{:.1f}\" y1=\"{:.1f}\" x2=\"{:.1f}\" y2=\"{:.1f}\" "
412 "stroke=\"{}\" stroke-width=\"4.0\"/>\n",
413 lx - 40.0, ly, lx - 10.0, ly, it->second);
414 svg << std::format(
415 " <path d=\"{}\" fill=\"{}\" stroke=\"none\"/>\n",
416 Roboto::instance().render(it->first, lx - 45.0, ly + 6.0, 18, "end"),
417 config.text_color());
418 ly += 28.0;
419 }
420
421 // Titles/Labels
422 svg << std::format(" <path d=\"{}\" fill=\"{}\" stroke=\"none\"/>\n",
423 Roboto::instance().render(x_label, (px0 + px1) / 2.0,
424 py1 + 75.0, 28, "middle"),
425 config.text_color());
426
427 // Y label rotated
428 svg << std::format(
429 " <g transform=\"translate({:.1f}, {:.1f}) rotate(-90)\">\n", 40.0,
430 (py0 + py1) / 2.0);
431 svg << std::format(" <path d=\"{}\" fill=\"{}\" stroke=\"none\"/>\n",
432 Roboto::instance().render(y_label, 0, 0, 28, "middle"),
433 config.text_color());
434 svg << " </g>\n";
435
436 svg << "</svg>\n";
437 return svg.str();
438}
