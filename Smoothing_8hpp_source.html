38inline void fillGaussian(std::vector<double> &kernel, const KernelGenerationParams &params, double center) {
39 const double prefactor = 1.0 / (std::sqrt(correlation::math::two_pi) * params.sigma);
40 const double exp_coeff = -1.0 / (2.0 * params.sigma * params.sigma);
41 for (size_t i = 0; i < params.size; ++i) {
42 const double distance = (static_cast<double>(i) - center) * params.bin_width;
43 kernel[i] = prefactor * std::exp(exp_coeff * distance * distance);
44 }
45}

47inline void fillTriweight(std::vector<double> &kernel, const KernelGenerationParams &params, double center) {
48 constexpr double factor = 35.0 / 32.0;
49 for (size_t i = 0; i < params.size; ++i) {
50 const double norm_dist = ((static_cast<double>(i) - center) * params.bin_width) / params.sigma;
51 if (std::abs(norm_dist) <= 1.0) {
52 const double norm_dist_sq = norm_dist * norm_dist;
53 kernel[i] = factor * std::pow(1.0 - norm_dist_sq, 3);
54 } else {
55 kernel[i] = 0.0;
56 }
57 }
58}

60inline void fillBump(std::vector<double> &kernel, const KernelGenerationParams &params, double center) {

116[[nodiscard]] inline std::vector<double> generateKernel(const KernelGenerationParams &params) {
117 const size_t size = params.size;
118 std::vector<double> kernel(size);
119 const double center = static_cast<double>(size - 1) / 2.0;
120
121 switch (params.type) {
122 case KernelType::Gaussian:
123 detail::fillGaussian(kernel, params, center);
124 break;
125 case KernelType::Triweight:
126 detail::fillTriweight(kernel, params, center);
127 break;
128 case KernelType::Bump:
129 detail::fillBump(kernel, params, center);
130 break;
131 case KernelType::Epanechnikov:
132 detail::fillEpanechnikov(kernel, params, center);
133 break;
134 case KernelType::Cosine:
135 detail::fillCosine(kernel, params, center);
136 break;
137 case KernelType::Biweight:
138 detail::fillBiweight(kernel, params, center);
139 break;
140 default:
141 throw std::invalid_argument("Unsupported kernel type for smoothing.");
142 }
143
144 // Normalize the discrete kernel to ensure the sum of its elements is 1.
145 const double sum = std::accumulate(kernel.begin(), kernel.end(), 0.0);
146 if (sum > 1e-9) {
147 for (double &val : kernel) {
148 val /= sum;
149 }
150 }
151 return kernel;
152}

168inline std::vector<double> KernelSmoothing(double bin_width, const std::vector<double> &y_values, double sigma,
169 KernelType type) {
170 if (y_values.empty()) {
171 return {};
172 }
173 if (bin_width <= 0) {
174 throw std::invalid_argument("'bin_width' must be a positive bin width for smoothing.");
175 }
176 if (sigma <= 0.0) {
177 throw std::invalid_argument("Smoothing sigma must be positive.");
178 }
179
180 // Calculate the kernel radius in bins by dividing the physical sigma by bin_width.
181 // We use 4.0 * sigma to cover approximately 99.99% of the Gaussian area.
182 const size_t num_points = y_values.size();
183 auto kernel_radius_bins = static_cast<size_t>(4.0 * sigma / bin_width);
184
185 // Clamp the radius to prevent the kernel from exceeding half the data size.
186 const size_t max_kernel_radius = num_points / 2;
187 size_t kernel_radius = std::min(kernel_radius_bins, max_kernel_radius);
188
189 // Ensure a minimum kernel size (3 bins: -1, 0, +1) if sigma is very small.
190 if (kernel_radius == 0) {
191 kernel_radius = 1;
192 }
193
194 const size_t kernel_size = 2 * kernel_radius + 1;
195
196 // Pass the original physical sigma (in distance units) to the kernel
197 // generator.
198 const auto kernel = generateKernel({kernel_size, bin_width, sigma, type});
199 std::vector<double> smoothed(num_points, 0.0);
200
201 // Phase 1 – left boundary (bins 0 .. kernel_radius-1): needs index clamping.
202 for (size_t i = 0; i < std::min(kernel_radius, num_points); ++i) {
203 for (size_t j = 0; j < kernel_size; ++j) {
204 const long long idx =
205 static_cast<long long>(i) + static_cast<long long>(j) - static_cast<long long>(kernel_radius);
206 smoothed[i] += y_values[static_cast<size_t>(std::max(0LL, idx))] * kernel[j];
207 }
208 }
209
210 // Phase 2 – interior (bins kernel_radius .. num_points-kernel_radius-1):
211 // index i - kernel_radius + j is always in [0, num_points-1], no branch needed.
212 // The clean inner loop (sum += src[j] * kernel[j]) is auto-vectorised by
213 // GCC/Clang at -O2 using whatever SIMD width the TU was compiled with.
214 if (num_points > 2 * kernel_radius) {
215 for (size_t i = kernel_radius; i < num_points - kernel_radius; ++i) {
216 const double *src = y_values.data() + (i - kernel_radius);
217 double sum = 0.0;
218 for (size_t j = 0; j < kernel_size; ++j) {
219 sum += src[j] * kernel[j];
220 }
221 smoothed[i] = sum;
222 }
223 }
224
225 // Phase 3 – right boundary (bins num_points-kernel_radius .. num_points-1): needs clamping.
226 const size_t right_start = (num_points > kernel_radius) ? num_points - kernel_radius : num_points;
227 for (size_t i = right_start; i < num_points; ++i) {
228 for (size_t j = 0; j < kernel_size; ++j) {
229 const long long idx =
230 static_cast<long long>(i) + static_cast<long long>(j) - static_cast<long long>(kernel_radius);
231 const long long clamped = std::min(idx, static_cast<long long>(num_points) - 1);
232 smoothed[i] += y_values[static_cast<size_t>(clamped)] * kernel[j];
233 }
234 }
235
236 return smoothed;
237}
