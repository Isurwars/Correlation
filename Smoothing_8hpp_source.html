35inline std::vector<double> generateKernel(size_t size, double dx, double sigma,
36 KernelType type) {
37 std::vector<double> kernel(size);
38 // Center of the discrete kernel in bin index units.
39 const double center = static_cast<double>(size - 1) / 2.0;
40
41 if (type == KernelType::Gaussian) {
42 // 1 / (sigma * sqrt(2*pi))
43 const double a = 1.0 / (std::sqrt(correlation::math::two_pi) * sigma);
44 const double b = -1.0 / (2.0 * sigma * sigma);
45 for (size_t i = 0; i < size; ++i) {
46 // Calculate x in distance units (r units)
47 const double x = (static_cast<double>(i) - center) * dx;
48 kernel[i] = a * std::exp(b * x * x);
49 }
50 } else if (type == KernelType::Triweight) {
51 constexpr double factor = 35.0 / 32.0;
52 for (size_t i = 0; i < size; ++i) {
53 // Calculate u (distance normalized by sigma)
54 const double u = ((static_cast<double>(i) - center) * dx) / sigma;
55 if (std::abs(u) <= 1.0) {
56 const double u2 = u * u;
57 kernel[i] = factor * std::pow(1 - u2, 3);
58 } else {
59 kernel[i] = 0.0;
60 }
61 }
62 } else if (type == KernelType::Bump) {
63 for (size_t i = 0; i < size; ++i) {
64 // Calculate u (distance normalized by sigma)
65 const double u = ((static_cast<double>(i) - center) * dx) / sigma;
66 if (std::abs(u) < 1.0) {
67 const double u2 = u * u;
68 kernel[i] = std::exp(-1.0 / (1.0 - u2));
69 } else {
70 kernel[i] = 0.0;
71 }
72 }
73 } else {
74 throw std::invalid_argument("Unsupported kernel type for smoothing.");
75 }
76
77 // Normalize the discrete kernel to ensure the sum of its elements is 1.
78 const double sum = std::accumulate(kernel.begin(), kernel.end(), 0.0);
79 if (sum > 1e-9) {
80 for (double &val : kernel) {
81 val /= sum;
82 }
83 }
84 return kernel;
85}

101inline std::vector<double> KernelSmoothing(double dx,
102 const std::vector<double> &y,
103 double sigma, KernelType type) {
104 if (y.empty()) {
105 return {};
106 }
107 if (dx <= 0) {
108 throw std::invalid_argument(
109 "'dx' must be a positive bin width for smoothing.");
110 }
111 if (sigma <= 0.0) {
112 throw std::invalid_argument("Smoothing sigma must be positive.");
113 }
114
115 // Calculate the kernel radius in bins by dividing the physical sigma by dx.
116 // We use 4.0 * sigma to cover approximately 99.99% of the Gaussian area.
117 const size_t n = y.size();
118 size_t kernel_radius_bins = static_cast<size_t>(4.0 * sigma / dx);
119
120 // Clamp the radius to prevent the kernel from exceeding half the data size.
121 const size_t max_kernel_radius = n / 2;
122 size_t kernel_radius = std::min(kernel_radius_bins, max_kernel_radius);
123
124 // Ensure a minimum kernel size (3 bins: -1, 0, +1) if sigma is very small.
125 if (kernel_radius == 0) {
126 kernel_radius = 1;
127 }
128
129 const size_t kernel_size = 2 * kernel_radius + 1;
130
131 // Pass the original physical sigma (in distance units) to the kernel
132 // generator.
133 const auto kernel = generateKernel(kernel_size, dx, sigma, type);
134 std::vector<double> smoothed(n, 0.0);
135
136 // Phase 1 – left boundary (bins 0 .. kernel_radius-1): needs index clamping.
137 for (size_t i = 0; i < std::min(kernel_radius, n); ++i) {
138 for (size_t j = 0; j < kernel_size; ++j) {
139 const long long idx = static_cast<long long>(i) +
140 static_cast<long long>(j) -
141 static_cast<long long>(kernel_radius);
142 smoothed[i] += y[static_cast<size_t>(std::max(0LL, idx))] * kernel[j];
143 }
144 }
145
146 // Phase 2 – interior (bins kernel_radius .. n-kernel_radius-1):
147 // index i - kernel_radius + j is always in [0, n-1], no branch needed.
148 // The clean inner loop (sum += src[j] * kernel[j]) is auto-vectorised by
149 // GCC/Clang at -O2 using whatever SIMD width the TU was compiled with.
150 if (n > 2 * kernel_radius) {
151 for (size_t i = kernel_radius; i < n - kernel_radius; ++i) {
152 const double *src = y.data() + (i - kernel_radius);
153 double sum = 0.0;
154 for (size_t j = 0; j < kernel_size; ++j)
155 sum += src[j] * kernel[j];
156 smoothed[i] = sum;
157 }
158 }
159
160 // Phase 3 – right boundary (bins n-kernel_radius .. n-1): needs clamping.
161 const size_t right_start = (n > kernel_radius) ? n - kernel_radius : n;
162 for (size_t i = right_start; i < n; ++i) {
163 for (size_t j = 0; j < kernel_size; ++j) {
164 const long long idx = static_cast<long long>(i) +
165 static_cast<long long>(j) -
166 static_cast<long long>(kernel_radius);
167 const long long clamped = std::min(idx, static_cast<long long>(n) - 1);
168 smoothed[i] += y[static_cast<size_t>(clamped)] * kernel[j];
169 }
170 }
171
172 return smoothed;
173}
