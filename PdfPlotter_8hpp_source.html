29// Extract color from string "#RRGGBB"
30inline uint32_t parseHexColor(const std::string &hex) {
31 if (hex.length() == 7 && hex[0] == '#') {
32 uint32_t red = std::stoul(hex.substr(1, 2), nullptr, 16);
33 uint32_t green = std::stoul(hex.substr(3, 2), nullptr, 16);
34 uint32_t blue = std::stoul(hex.substr(5, 2), nullptr, 16);
35 return PDF_RGB(red, green, blue);
36 }
37 return PDF_BLACK;
38}

45// Blend foreground color with background color using a given opacity to simulate transparency
46inline uint32_t blendColor(BlendParams colors, double opacity) {
47 double r_fg = ((colors.fg >> 16) & 0xff);
48 double g_fg = ((colors.fg >> 8) & 0xff);
49 double b_fg = (colors.fg & 0xff);
50
51 double r_bg = ((colors.bg >> 16) & 0xff);
52 double g_bg = ((colors.bg >> 8) & 0xff);
53 double b_bg = (colors.bg & 0xff);
54
55 uint32_t red = static_cast<uint32_t>(r_fg * opacity + r_bg * (1.0 - opacity));
56 uint32_t green = static_cast<uint32_t>(g_fg * opacity + g_bg * (1.0 - opacity));
57 uint32_t blue = static_cast<uint32_t>(b_fg * opacity + b_bg * (1.0 - opacity));
58
59 return PDF_RGB(red, green, blue);
60}

62// Format numbers in standard ASCII scientific notation for PDF compatibility
63inline std::string fmtScientificPdf(double value) {
64 double abs_value = std::abs(value);
65 if (abs_value < 1e-12) {
66 return "0";
67 }
68
69 if (abs_value < 0.001 || abs_value >= 10000.0) {
70 int exponent = static_cast<int>(std::floor(std::log10(abs_value)));
71 double fraction = value / std::pow(10.0, exponent);
72 return std::format("{:.1f}x10^{}", fraction, exponent);
73 }
74
75 std::string str = std::format("{:.2f}", value);
76 auto dot = str.find('.');
77 if (dot != std::string::npos) {
78 std::size_t last = str.find_last_not_of('0');
79 if (last != std::string::npos && last > dot) {
80 str = str.substr(0, last + 1);
81 } else if (last == dot) {
82 str = str.substr(0, dot);
83 }
84 }
85 return str;
86}

88// Sanitize unit strings to use ASCII superscripts instead of Unicode
89inline std::string sanitizeUnitPdf(const std::string &unit) {
90 std::string clean = unit;
91 auto replaceAll = [](std::string &str, const std::string &search_for, const std::string &replacement) {
92 size_t start_pos = 0;
93 while ((start_pos = str.find(search_for, start_pos)) != std::string::npos) {
94 str.replace(start_pos, search_for.length(), replacement);
95 start_pos += replacement.length();
96 }
97 };
98 replaceAll(clean, "⁻¹", "^-1");
99 replaceAll(clean, "⁻", "^-");
100 replaceAll(clean, "⁰", "^0");
101 replaceAll(clean, "¹", "^1");
102 replaceAll(clean, "²", "^2");
103 replaceAll(clean, "³", "^3");
104 replaceAll(clean, "⁴", "^4");
105 replaceAll(clean, "⁵", "^5");
106 replaceAll(clean, "⁶", "^6");
107 replaceAll(clean, "⁷", "^7");
108 replaceAll(clean, "⁸", "^8");
109 replaceAll(clean, "⁹", "^9");
110 replaceAll(clean, "×", "x");
111 return clean;
112}

119// Helper to draw text with specified alignment (TextAnchor) and optional rotation angle
120inline void drawPdfText(pdf_doc *pdf, struct pdf_object *page, const std::string &text, PdfPoint pos, double size,
121 TextAnchor anchor, uint32_t color, double angle = 0.0) {
122 std::string sanitized = sanitizeUnitPdf(text);
123 float width = 0.0F;
124 pdf_get_font_text_width(pdf, "Helvetica", sanitized.c_str(), static_cast<float>(size), &width);
125
126 double offset_factor = 0.0;
127 if (anchor == TextAnchor::Middle) {
128 offset_factor = 0.5;
129 } else if (anchor == TextAnchor::End) {
130 offset_factor = 1.0;
131 }
132
133 double offset_x = -offset_factor * width * std::cos(angle);
134 double offset_y = -offset_factor * width * std::sin(angle);
135 double x_aligned = pos.x + offset_x;
136 double y_aligned = pos.y + offset_y;
137
138 if (angle != 0.0) {
139 pdf_add_text_rotate(pdf, page, sanitized.c_str(), static_cast<float>(size), static_cast<float>(x_aligned),
140 static_cast<float>(y_aligned), static_cast<float>(angle), color);
141 } else {
142 pdf_add_text(pdf, page, sanitized.c_str(), static_cast<float>(size), static_cast<float>(x_aligned),
143 static_cast<float>(y_aligned), color);
144 }

648inline void renderHistogramAsPdf(const correlation::analysis::Histogram &hist, const std::string &filepath,
649 const PlotConfig &config) {
650 double canvas_width = config.effective_width();
651 double canvas_height = config.effective_height();
652
653 // Setup pdfgen
654 pdf_info info = {};
655 pdf_doc *pdf = pdf_create(static_cast<float>(canvas_width), static_cast<float>(canvas_height), &info);
656 struct pdf_object *page = pdf_append_page(pdf);
657
658 uint32_t bg_col = detail::parseHexColor(config.bg_color());
659 uint32_t text_col = detail::parseHexColor(config.text_color());
660
661 // Background
662 pdf_add_filled_rectangle(pdf, page, 0.0F, 0.0F, static_cast<float>(canvas_width), static_cast<float>(canvas_height), 0.0F, bg_col,
663 PDF_TRANSPARENT);
664
665 pdf_set_font(pdf, "Helvetica");
666
667 auto toPdfY = [&](double y_svg) { return canvas_height - y_svg; };
668
669 if (hist.bins.empty() || (hist.smoothed_partials.empty() && hist.partials.empty())) {
670 detail::drawPdfText(pdf, page, "No data available", {canvas_width / 2.0, toPdfY(canvas_height / 2.0)}, 24.0 * config.font_scale,
671 TextAnchor::Middle, text_col);
672 pdf_save(pdf, filepath.c_str());
673 pdf_destroy(pdf);
674 return;
675 }
676
677 detail::PdfHistogramRenderer renderer(hist, config, pdf, page);
678 if (renderer.prepareData()) {
679 renderer.drawGridAndTicks();
680 renderer.drawEmphasisLines();
681 renderer.drawAreaFills();
682 auto legend_items = renderer.drawPolylines();
683 renderer.drawMarkers();
684 renderer.drawLegend(legend_items);
685 renderer.drawTitlesAndLabels();
686 }
687
688 pdf_save(pdf, filepath.c_str());
689 pdf_destroy(pdf);
690}
